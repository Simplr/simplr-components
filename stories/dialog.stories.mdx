# Simplr Dialog

### Usage

```js
import SimplrDialog from '@simplr-wc/dialog';
import '@simplr-wc/button';

const openDialog = () => {
  const dialog = new SimplrDialog({
      primaryAction: {label: "Cool"},
      secondaryAction: {label: "Bleh"},
      heading: "Hello",
      content: "Hello world :) This is my first dialog"
  });
  dialog.openDialog();
}

```

```html
<simplr-button
 @click=${() => openDialog()}
 raised
 primary
 label="Open dialog"
></simplr-button>
```

##### Action structure

```js
// Example action
const saveDocument = () => {
  myDoc.save();
}

const primaryAction = {
  label: "Save document",
  action: saveDocument
};

const dialog = new SimplrDialog({
    primaryAction: primaryAction,
    secondaryAction: {label: "No"},
    heading: "Save Document",
    content: "Would you like to save the document?"
});

```
