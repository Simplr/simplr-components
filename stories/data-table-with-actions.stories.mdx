# Simplr Data Table with actions

### Usage

```js
import '@simplr-wc/data-table';

const tableOptions = {
    isRowSelectable: true,
    tableClass: "my-data-table"
};

const tableColumns = ["First name", "Last name", "Gender", "Actions"];

const tableActions = [ 
    {icon: 'cancel', type: 'button', color: '#F08080', onClick: (rowData) => console.log('Decline action on row', rowData)},
    {icon: 'done', type: 'button', color: '#5dbb61', onClick: (rowData) => console.log('Approve action on row', rowData)}
];

const tableData = [
    {id: 1, data: ["Bob", "The Builder", "Male", tableActions]},
    {id: 2, data: ["Joe", "Exotic", "Tiger King", tableActions]}
];
```

```html
<simplr-data-table
    @rowClick=${e => console.log(e)}
    tableLabel="My data table"
    .tableOptions=${tableOptions}
    .tableColumns=${tableColumns}
    .tableData=${tableData}
></simplr-data-table>
```
